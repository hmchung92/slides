Install Hadoop

  1. Install Java 
    $ brew install java
  2. Enable SSH 
    $ ssh-keygen -t rsa
$ cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
  3. Install Hadoop 
    $ brew install hadoop # Install in /usr/local/Cellar/hadoop/
  4. Add this to /libexec/conf/hadoop-env.sh 
    export HADOOP_OPTS="-Djava.security.krb5.realm=-Djava.security.krb5.kdc="
  5. Format hadoop nodename 
    $ bin/hadoop namenode -format
  6. Start/Stop 
    $ bin/start-all.sh             # Start hadoop
$ jps                          # List Java Processes
$ bin/stop-all.sh              # Stop hadoop

Data-binding programmatically

> bin/hadoop fs -ls                             # List files/dirs in HDFS
> bin/hadoop fs -mkdir                          # Make directory in HDFS
> bin/hadoop fs -copyFromLocal                  # Copy from local file system to HDFS
> bin/hadoop fs -copyToLocal                    # Copy from HDFS to local file system
> bin/hadoop fs -moveToLocal                    # Move from HDFS to local file system
> bin/hadoop fs -rm                             # Delete file
> bin/hadoop fs -rmr                            # Delete directory
> bin/hadoop fs -tail                           # Tail files
> bin/hadoop fs -chmod                          # Change permissions
> bin/hadoop fs -setrep -w 4 -R /dir1/s-dir/    # Change replication factor to 4 for all files in the given subdirectory
